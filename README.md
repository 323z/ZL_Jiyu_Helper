#include <iostream> #include <fstream> #include <string> #include <vector> #include <cstdlib> #include <cstring> #include <csignal> #include <chrono> #include <thread> #include <dirent.h> #include <signal.h> #include <unistd.h> #include <sys/stat.h> // 全局标志：控制循环是否继续运行 volatile sig_atomic_t g_running = 1; // 信号处理函数：捕获Ctrl+C，优雅退出 void signal_handler(int signum) { if (signum == SIGINT) { std::cout << "\n\n接收到退出信号，正在停止持续监控..." << std::endl; g_running = 0; } } // 读取进程的纯进程名（从/proc/[pid]/comm读取，精准无参数） std::string get_process_name(pid_t pid) { std::string comm_path = "/proc/" + std::to_string(pid) + "/comm"; std::ifstream file(comm_path); if (!file.is_open()) { return ""; } std::string proc_name; std::getline(file, proc_name); // comm文件仅一行，存储纯进程名 file.close(); return proc_name; } // 精准查找进程名完全等于target_name的进程 std::vector<pid_t> find_processes_by_exact_name(const std::string& target_name) { std::vector<pid_t> matched_pids; DIR* proc_dir = opendir("/proc"); if (!proc_dir) { std::cerr << "错误：无法打开/proc目录，请检查权限！" << std::endl; return matched_pids; } dirent* entry; while ((entry = readdir(proc_dir)) != nullptr && g_running) { // 只处理数字命名的目录（进程ID） if (entry->d_type != DT_DIR) { continue; } char* endptr; pid_t pid = strtol(entry->d_name, &endptr, 10); if (*endptr != '\0') { // 非数字目录跳过（非进程目录） continue; } // 跳过当前进程自身，避免自杀 if (pid == getpid()) { continue; } // 读取纯进程名（精准匹配的核心） std::string proc_name = get_process_name(pid); if (proc_name.empty()) { continue; } // 严格全等匹配：进程名必须完全等于"Student"，多/少字符都不匹配 if (proc_name == target_name) { matched_pids.push_back(pid); } } closedir(proc_dir); return matched_pids; } // 强制终止进程（kill -9，发送SIGKILL信号） bool kill_process(pid_t pid) { int ret = kill(pid, SIGKILL); if (ret == -1) { std::cerr << "[" << __TIME__ << "] 错误：无法终止进程 " << pid << " (权限不足或进程已退出)" << std::endl; return false; } std::cout << "[" << __TIME__ << "] 成功终止进程：PID=" << pid << " 进程名=Student" << std::endl; return true; } int main() { const std::string target_process_name = "Student"; // 精准匹配的目标进程名 const bool continuous_mode = true; // 固定为true，开启持续查杀 const int scan_interval_seconds = 2; // 持续扫描间隔（可调整，如1/3/5秒） // 注册信号处理函数（捕获Ctrl+C） signal(SIGINT, signal_handler); // 检查运行权限（非root可能无法终止部分进程） if (getuid() != 0) { std::cerr << "警告：建议使用root权限运行（sudo），否则可能无法终止部分进程！" << std::endl; } // 输出程序说明 std::cout << "========================================" << std::endl; std::cout << "进程持续查杀程序：精准匹配进程名=\"" << target_process_name << "\"" << std::endl; std::cout << "扫描间隔：" << scan_interval_seconds << "秒" << std::endl; std::cout << "按 Ctrl+C 退出程序" << std::endl; std::cout << "========================================" << std::endl; // 核心逻辑：持续循环查杀 while (g_running) { // 1. 查找精准匹配的进程 std::vector<pid_t> matched_pids = find_processes_by_exact_name(target_process_name); // 2. 处理匹配结果 if (!matched_pids.empty()) { std::cout << "\n[" << __TIME__ << "] 发现" << matched_pids.size() << "个精准匹配的进程，开始终止..." << std::endl; for (pid_t pid : matched_pids) { kill_process(pid); } } // 3. 休眠指定间隔后再次扫描（避免CPU占用过高） if (g_running) { std::this_thread::sleep_for(std::chrono::seconds(scan_interval_seconds)); } } std::cout << "持续查杀程序已正常退出" << std::endl; return 0; }
